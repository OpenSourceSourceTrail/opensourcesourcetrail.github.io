329 static void addTo(Branch* branch, const StringT& str, size_t idx, const ValueT& val) {
330 CharT c(0);
331 if(idx < str.size()) {
332 c = str[idx];
333 }
334
335 auto it = branch->m_children.find(c);
336 if(it == branch->m_children.end()) {
337 size_t newIdx = idx + 1 >= str.size() ? str.size() : idx + 1;
338 branch->m_children.emplace(c, branch->createLeaf(str.substr(newIdx), val));
339 return;
340 }
341
342 idx++;
343
344 std::unique_ptr<Node> child = std::move(it->second);
345 StringT childStr = child->getString();
346
347 auto p = std::mismatch(childStr.begin(), childStr.end(), str.begin() + idx, str.end());
348
349 if(p.first == childStr.end() && p.second == str.end()) {
350 // adding same string, abort
351 it->second = std::move(child);
352 return;
353 }
354
355 size_t length = std::distance(childStr.begin(), p.first);
356
357 std::unique_ptr<Branch> newBranch;
358 if(!dynamic_cast<Branch*>(child.get()) || p.first != childStr.end()) {
359 newBranch = branch->split(std::move(child), length);
360 } else {
361 newBranch = std::unique_ptr<Branch>(dynamic_cast<Branch*>(child.release()));
362 }
363
364 addTo(newBranch.get(), str, idx + length, val);
365
366 it->second = std::move(newBranch);
367 }
368
369 size_t getByteSize() const override {
370 size_t s = m_children.size() * sizeof(std::pair<CharT, std::unique_ptr<Node>>);
371
372 for(const auto& p : m_children) {
373 s += p.second->getByteSize();
374 }
375
376 return s;
377 }
378
379 virtual ValueT find(const StringT& str, size_t idx) const override {
380 if(idx > str.size()) {
381 return defaultVal;
382 }
383
384 std::pair<bool, size_t> p = this->compareString(str, idx);
385 if(!p.first) {
386 return defaultVal;
387 }
388 idx += p.second;
389
390 CharT c(0);
391 if(idx < str.size()) {
392 c = str[idx];
393 idx++;
394 }
395
396 auto it = m_children.find(c);
397 if(it != m_children.end()) {
398 return it->second->find(str, idx);
399 }
400
401 return defaultVal;
402 }

404 void print(StreamT& os, size_t depth) const override {
405 StringT myStr = this->getString();
406 if(myStr.size()) {
407 os << StringT(depth, ' ') << myStr << std::endl;
408 depth += myStr.size();
409 }
410
411 for(const auto& p : m_children) {
412 os << StringT(depth, ' ') << '|' << p.first << '|' << std::endl;
413 p.second->print(os, depth + 3);
414 }
415 }

421 std::unique_ptr<Branch> split(std::unique_ptr<Node> node, size_t idx) const {
422 StringT str = node->getString();
423 std::unique_ptr<Branch> frontBranch = createBranch(str.substr(0, idx));
424
425 CharT c(0);
426 if(idx < str.size()) {
427 c = str[idx];
428 idx++;
429 }
430
431 Branch* oldBranch = dynamic_cast<Branch*>(node.get());
432 if(oldBranch) {
433 std::unique_ptr<Branch> backBranch = createBranch(str.substr(idx));
434 backBranch->m_children = std::move(oldBranch->m_children);
435 frontBranch->m_children.emplace(c, std::move(backBranch));
436 } else {
437 frontBranch->m_children.emplace(c, createLeaf(str.substr(idx), node->getValue()));
438 }
439
440 return frontBranch;
441 }
